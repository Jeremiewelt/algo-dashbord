<script>

const PASSWORD = ‘Rocket’;

const CAPITAL_INITIAL = 3000;


function checkPassword() {

  const input = document.getElementById('password-input').value;

  if (input === PASSWORD) {

    sessionStorage.setItem('auth', 'true');

    showDashboard();

  } else {

    document.getElementById('error-msg').style.display = 'block';

  }

}


function logout() {

  sessionStorage.removeItem('auth');

  document.getElementById('dashboard').style.display = 'none';

  document.getElementById('login-screen').style.display = 'flex';

  document.getElementById('password-input').value = '';

}


function showDashboard() {

  document.getElementById('login-screen').style.display = 'none';

  document.getElementById('dashboard').style.display = 'block';

  loadData();

  fetchBTCPrice();

  setInterval(fetchBTCPrice, 60000);

}


async function fetchBTCPrice() {

  try {

    const r = await fetch('https://api.binance.com/api/v3/ticker/price?symbol=BTCUSDT');

    const d = await r.json();

    const price = parseFloat(d.price).toLocaleString('fr-FR', {maximumFractionDigits: 0});

    document.getElementById('btc-price').textContent = 'BTC: ' + price + '$';

  } catch(e) {}

}


function loadData() {

  const trades = JSON.parse(localStorage.getItem('trades') || '[]');

  updateMetrics(trades);

  updateChart(trades);

  updateTradesTable(trades);

}


function updateMetrics(trades) {

  const capital = CAPITAL_INITIAL + trades.reduce((s, t) => s + t.pnl, 0);

  const perf = ((capital - CAPITAL_INITIAL) / CAPITAL_INITIAL * 100).toFixed(1);

  const pnl = capital - CAPITAL_INITIAL;

  const wins = trades.filter(t => t.pnl > 0).length;

  const wr = trades.length > 0 ? Math.round(wins / trades.length * 100) : 0;

  const best = trades.length > 0 ? Math.max(...trades.map(t => t.pnl)) : 0;

  const worst = trades.length > 0 ? Math.min(...trades.map(t => t.pnl)) : 0;


  let runningCapital = CAPITAL_INITIAL;

  let peak = CAPITAL_INITIAL;

  let maxDD = 0;

  trades.forEach(t => {

    runningCapital += t.pnl;

    if (runningCapital > peak) peak = runningCapital;

    const dd = (runningCapital - peak) / peak * 100;

    if (dd < maxDD) maxDD = dd;

  });


  document.getElementById('m-capital').textContent = Math.round(capital).toLocaleString('fr-FR') + '€';

  document.getElementById('m-perf').textContent = (perf >= 0 ? '+' : '') + perf + '%';

  document.getElementById('m-perf').className = 'metric-sub ' + (perf >= 0 ? 'positive' : 'negative');

  document.getElementById('m-pnl').textContent = (pnl >= 0 ? '+' : '') + Math.round(pnl) + '€';

  document.getElementById('m-pnl').className = 'metric-value ' + (pnl >= 0 ? 'positive' : 'negative');

  document.getElementById('m-winrate').textContent = trades.length > 0 ? wr + '%' : '—';

  document.getElementById('m-trades').textContent = trades.length + ' trades';

  document.getElementById('m-best').textContent = '+' + Math.round(best) + '€';

  document.getElementById('m-worst').textContent = Math.round(worst) + '€';

  document.getElementById('m-drawdown').textContent = maxDD.toFixed(1) + '%';


  document.getElementById('a1-pnl').textContent = (pnl >= 0 ? '+' : '') + Math.round(pnl) + '€';

  document.getElementById('a1-wr').textContent = trades.length > 0 ? wr + '%' : '—';

  document.getElementById('a1-trades').textContent = trades.length;

  document.getElementById('a1-capital').textContent = Math.round(capital) + '€';

  document.getElementById('a1-progress').style.width = Math.min(wr, 100) + '%';

}


function updateChart(trades) {

  const labels = [];

  const data = [];

  let capital = CAPITAL_INITIAL;

  labels.push('Début');

  data.push(capital);

  trades.forEach(t => {

    capital += t.pnl;

    const date = new Date(t.date);

    labels.push(date.toLocaleDateString('fr-FR', {day:'2-digit', month:'2-digit'}));

    data.push(Math.round(capital));

  });


  if (window.perfChart) window.perfChart.destroy();

  const ctx = document.getElementById('perf-chart').getContext('2d');

  window.perfChart = new Chart(ctx, {

    type: 'line',

    data: {

      labels,

      datasets: [

        {

          label: 'Capital',

          data,

          borderColor: '#22c55e',

          backgroundColor: 'rgba(34,197,94,0.08)',

          borderWidth: 2,

          pointRadius: 3,

          fill: true,

          tension: 0.4

        },

        {

          label: 'Initial',

          data: Array(labels.length).fill(CAPITAL_INITIAL),

          borderColor: '#3b82f6',

          backgroundColor: 'transparent',

          borderWidth: 1,

          borderDash: [4, 4],

          pointRadius: 0

        }

      ]

    },

    options: {

      responsive: true,

      maintainAspectRatio: false,

      plugins: { legend: { display: false } },

      scales: {

        x: { grid: { color: '#1a1a1a' }, ticks: { color: '#555', font: { size: 11 } } },

        y: { grid: { color: '#1a1a1a' }, ticks: { color: '#555', font: { size: 11 }, callback: v => v + '€' } }

      }

    }

  });

}


function updateTradesTable(trades) {

  const tbody = document.getElementById('trades-body');

  if (trades.length === 0) {

    tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:#555;padding:24px;">Aucun trade enregistré</td></tr>';

    return;

  }

  const recent = [...trades].reverse().slice(0, 20);

  tbody.innerHTML = recent.map(t => {

    const date = new Date(t.date).toLocaleDateString('fr-FR', {day:'2-digit', month:'2-digit', hour:'2-digit', minute:'2-digit'});

    const pnlColor = t.pnl >= 0 ? '#22c55e' : '#ef4444';

    const pnlStr = (t.pnl >= 0 ? '+' : '') + Math.round(t.pnl) + '€';

    return `<tr>

      <td style="color:#666">${date}</td>

      <td><span class="pill ${t.sens === 'BUY' ? 'pill-buy' : 'pill-sell'}">${t.sens}</span></td>

      <td><span class="pill ${t.regime === 'range' ? 'pill-range' : 'pill-tendance'}">${t.regime}</span></td>

      <td style="color:#888">${t.entree ? Math.round(t.entree).toLocaleString('fr-FR') + '$' : '—'}</td>

      <td style="color:#888">${t.sortie ? Math.round(t.sortie).toLocaleString('fr-FR') + '$' : '—'}</td>

      <td><span class="pill ${t.raison === 'take_profit' ? 'pill-tp' : 'pill-stop'}">${t.raison}</span></td>

      <td style="text-align:right;font-weight:500;color:${pnlColor}">${pnlStr}</td>

    </tr>`;

  }).join('');

}


if (sessionStorage.getItem('auth') === 'true') {

  showDashboard();

}
